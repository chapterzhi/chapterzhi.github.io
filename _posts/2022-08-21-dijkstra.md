class MyCode {

    int n = 5;
    int[][] matrix = new int[n][n];

    MyCode() {
        for (int[] arr : matrix) {
            Arrays.fill(arr, Integer.MAX_VALUE);
        }

        for (int i = 0; i < matrix.length; i++) {
            for(int j = 0; j < matrix[0].length; j++) {
                if (i == j) {
                    matrix[i][j] = 0;
                }
            }
        }

        matrix[0][1] = 10;
        matrix[0][3] = 30;
        matrix[0][4] = 100;
        matrix[1][2] = 50;
        matrix[2][4] = 10;
        matrix[3][2] = 20;
        matrix[3][4] = 60;
    }

    void dijkstra(Integer v) {
        // 最短路径节点集
        boolean s[] = new boolean[n]; // 已经找到最短路径的集合
        int[] dist = new int[n];      // 暂存的最短路径
        int[] path = new int[n];      // 到某点的路径（记录的是上一个节点）

        for (int i = 0; i < n; i++) {
            dist[i] = matrix[v][i];
            s[i] = false;
            if (i != v && dist[i] != Integer.MAX_VALUE) {
                path[i] = v;
            } else {
                path[i] = -1;
            }
        }

        s[v] = true;
        for (int i = 0; i < n - 1; i++) {
            Integer min = Integer.MAX_VALUE, u = v;
            for (int j = 0; j < n; j++) {
                if (!s[j] && dist[j] < min) {
                    u = j;
                    min = dist[j];
                }
            }
            s[u] = true;
            for (int k = 0; k < n; k++) {
                int w = matrix[u][k];
                if (!s[k] && w < Integer.MAX_VALUE && dist[u] + w < dist[k]) {
                    dist[k] = dist[u] + w;
                    path[k] = u;
                }
            }
        }

        System.out.println(Arrays.toString(path));
    }
}
