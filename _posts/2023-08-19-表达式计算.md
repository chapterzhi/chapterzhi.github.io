class Solution {

    public static void main(String[] args) {
        //     ( */ +- )
        // isp 1 5  3  6
        // icp 6 4  2  1
        Map<Character, int[]> map = new HashMap<>();
        map.put('(', new int[] {1, 6});
        map.put('*', new int[] {5, 4});
        map.put('/', new int[] {5, 4});
        map.put('+', new int[] {3, 2});
        map.put('-', new int[] {3, 2});
        map.put(')', new int[] {6, 1});
        map.put('#', new int[] {0, 0});
        
        String s = "1-(2+3)*6";
        s += '#';
        Stack<Character> stack = new Stack<>();
        stack.push('#');
        int ind = 0;
        StringBuffer output = new StringBuffer();
        while (s.charAt(ind) != '#' || stack.peek() != '#') {
            char c = s.charAt(ind);
            if (c >= '0' && c <= '9') {
                output.append(c);
                ind++;
            } else {
                    int icp = map.get(c)[1];
                    int topIsp = map.get(stack.peek())[0];
                    if (icp > topIsp) {
                        stack.push(c);
                        ind++;
                    } else if (icp < topIsp) {
                        output.append(stack.pop());
                    } else {
                        stack.pop();
                        ind++;
                    }
            }
        }
        System.out.println(output);
    }
}
