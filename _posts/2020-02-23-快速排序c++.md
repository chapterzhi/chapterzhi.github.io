'''c++
void myFastSort(vector<int>& nums,int low,int high) {
	if (!(low < high)) {
		return;
	}

	//分
	int midPos = low, midNum = nums[low];
	for (int i = low + 1; i <= high; i++) {
		if (nums[i] < midNum) {
			midPos++;
			if (i != midPos) {
				int t = nums[i];
				nums[i] = nums[midPos];
				nums[midPos] = t;
			}
		}
	}
	nums[low] = nums[midPos];
	nums[midPos] = midNum;

	// 排左
	myFastSort(nums, low, midPos - 1);

	// 排右
	myFastSort(nums, midPos + 1, high);
}
'''
